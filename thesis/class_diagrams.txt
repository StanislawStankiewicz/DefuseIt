    %% Definicja Protokołu Komunikacyjnego i Typów
    class Protocol {
        <<enumeration>>
        CMD_START_GAME = 0x01
        CMD_END_GAME = 0x02
        CMD_GET_STATUS = 0x03
        CMD_SET_STATUS = 0x04
        CMD_GET_VERSION = 0x05
        CMD_SET_VERSION = 0x06
        CMD_IDENTIFY = 0x07
        CMD_SET_REMAINING_SECONDS = 0x08
        CMD_SET_MISTAKE_COUNT = 0x09
        CMD_UNKNOWN = 0xFF
    }

    class Status {
        <<enumeration>>
        STATUS_UNSOLVED = 0x01
        STATUS_PASSED = 0x02
        STATUS_FAILED = 0x03
    }

    class FunctionPointers {
        <<typedef>>
        GameLoop : void*
        ResetState : void*
    }

    %% Klasa Master
    class Master {
        -uint8_t version
        -uint8_t moduleAddresses[11]
        -uint8_t moduleCount
        +Master(uint8_t version)
        +begin() void
        +discoverModules() void
        +startGame() void
        +endGame() void
        +restartFailedModule(uint8_t index) void
        +getModuleStatus(uint8_t index) uint8_t
        +setModuleStatus(uint8_t index, uint8_t status) void
        +sendVersion(uint8_t index) void
        +sendRemainingSeconds(uint8_t moduleAddress, uint16_t remainingSeconds) void
        +sendMistakeCount(uint8_t moduleAddress, uint8_t mistakeCount) void
        +broadcastMistakeCount(uint8_t mistakeCount) void
        +getVersion() uint8_t
        +setVersion(uint8_t newVersion) void
        +getModuleCount() uint8_t
        +getModuleAddress(uint8_t index) uint8_t
        +sendCommand(uint8_t moduleAddress, uint8_t command) void
    }

    %% Klasa Slave
    class Slave {
        -static uint8_t moduleAddress$
        -static uint8_t status$
        -static uint8_t masterVersion$
        -static uint16_t remainingSeconds$
        -static uint8_t mistakes$
        -static uint8_t command$
        -static uint8_t led_pin$
        -static bool isGameActive$
        -static GameLoop gameLoop$
        -static ResetState resetState$
        +Slave(uint8_t address, GameLoop loop, ResetState setState, uint8_t pin)
        +begin() void
        +getVersion() uint8_t
        +getRemainingSeconds() uint16_t
        +getMistakeCount() uint8_t
        +static startGame() void$
        +slaveLoop() void
        +pass() void
        +fail() void
        +static endGame() void$
        -static receiveEvent(int numBytes) void$
        -static requestEvent() void$
    }

    %% Relacje
    Master ..> Protocol : uses
    Slave ..> Protocol : uses
    Slave ..> Status : uses
    Slave ..> FunctionPointers : uses
    Master "1" -- "*" Slave : I2C Communication
